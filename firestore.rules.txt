rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    
    // 1. Users Collection: Users can ONLY read and update their own specific profile document
    match /users/{userId} {
      allow read, update, create: if request.auth != null && request.auth.uid == userId;
    }
    
    // 2. Config Collection: Anyone can read config 
    match /config/{document=**} {
      allow read: if true;
      allow write: if false; // Only admin can change from console
    }

    // 3. Submissions Collection
    match /submissions/{submissionId} {
      // Anyone can submit a new application
      allow create: if true; 
      
      // Users can only read/update their OWN submissions if logged in
      allow read, update, delete: if request.auth != null && request.auth.uid == resource.data.userId;
    }
    
    // 4. Default fallback (block everything else)
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
